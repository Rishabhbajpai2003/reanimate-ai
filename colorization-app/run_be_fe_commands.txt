#!/usr/bin/env bash
set -euo pipefail

# Run this file from anywhere; paths are resolved relative to this script.
ROOT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
BACKEND_DIR="$ROOT_DIR/backend"
FRONTEND_DIR="$ROOT_DIR/frontend"
SADTALKER_DIR="$ROOT_DIR/SadTalker"

BACKEND_PORT=5000
FRONTEND_PORT=3000

echo "==> Project root: $ROOT_DIR"

# =========================
# Kill old processes
# =========================
echo "==> Cleaning old processes..."
lsof -ti tcp:$BACKEND_PORT | xargs kill -9 2>/dev/null || true
lsof -ti tcp:$FRONTEND_PORT | xargs kill -9 2>/dev/null || true
pgrep -f "python app.py" | xargs kill -9 2>/dev/null || true

# =========================
# Backend setup
# =========================
echo "==> Backend setup"
cd "$BACKEND_DIR"

PY_RUN=()
PIP_RUN=()
INSTALL_MARKER="$BACKEND_DIR/.installed"

# Prefer active Conda/Micromamba env when available.
if [ -n "${CONDA_PREFIX:-}" ] && command -v python >/dev/null 2>&1; then
  echo "==> Using active Conda/Mamba environment: ${CONDA_DEFAULT_ENV:-$CONDA_PREFIX}"
  PY_RUN=(python)
  PIP_RUN=(python -m pip)
else
  if command -v python3.11 >/dev/null 2>&1; then
    BASE_PY=python3.11
  elif command -v python3 >/dev/null 2>&1; then
    BASE_PY=python3
  else
    echo "❌ ERROR: python3 not found"
    exit 1
  fi

  if [ ! -d ".venv" ]; then
      echo "==> Creating backend virtual environment"
      "$BASE_PY" -m venv .venv
  fi

  PY_RUN=("$BACKEND_DIR/.venv/bin/python")
  PIP_RUN=("$BACKEND_DIR/.venv/bin/pip")
fi

# =========================
# Install backend deps (ONCE)
# =========================
if [ ! -f "$INSTALL_MARKER" ]; then
  echo "==> Installing backend dependencies..."

  "${PIP_RUN[@]}" install --upgrade pip
  "${PIP_RUN[@]}" install wheel packaging
  "${PIP_RUN[@]}" install git+https://github.com/XPixelGroup/BasicSR.git@master
  "${PIP_RUN[@]}" install -r "$ROOT_DIR/requirements.txt"

  touch "$INSTALL_MARKER"
else
  echo "✓ Backend dependencies already installed"
fi

# =========================
# Models setup (NO RE-DOWNLOAD)
# =========================
echo "==> Setting up models..."
if [ ! -d "$BACKEND_DIR/models/BFM" ]; then
  "${PY_RUN[@]}" download_models.py --all
else
  echo "✓ Models already exist — skipping download"
fi

# =========================
# Clone repos if missing
# =========================
cd "$ROOT_DIR"
for repo in "CodeFormer" "SadTalker"; do
  if [ ! -d "$repo" ]; then
    echo "==> Cloning $repo"
    git clone "https://github.com/$( [ "$repo" == "CodeFormer" ] && echo "sczhou" || echo "OpenTalker" )/$repo.git"
  fi
done

# =========================
# SadTalker separate env (CRITICAL)
# =========================
echo "==> Setting up SadTalker environment"

if [ ! -d "$SADTALKER_DIR/sadtalker_env" ]; then
  python3.11 -m venv "$SADTALKER_DIR/sadtalker_env"

  source "$SADTALKER_DIR/sadtalker_env/bin/activate"
  pip install --upgrade pip
  pip install -r "$SADTALKER_DIR/requirements.txt"
  deactivate
else
  echo "✓ SadTalker env already exists"
fi

# =========================
# Link models (space saving)
# =========================
echo "==> Linking models"

mkdir -p "$SADTALKER_DIR/gfpgan/weights"
mkdir -p "$SADTALKER_DIR/checkpoints"
mkdir -p "$ROOT_DIR/CodeFormer/weights/CodeFormer"

ln -sf "$BACKEND_DIR/models/GFPGANv1.4.pth" "$SADTALKER_DIR/gfpgan/weights/" || true
ln -sf "$BACKEND_DIR/models/detection_Resnet50_Final.pth" "$SADTALKER_DIR/gfpgan/weights/" || true
ln -sf "$BACKEND_DIR/models/alignment_WFLW_4HG.pth" "$SADTALKER_DIR/gfpgan/weights/" || true
ln -sf "$BACKEND_DIR/models/parsing_parsenet.pth" "$SADTALKER_DIR/gfpgan/weights/" || true

ln -sf "$BACKEND_DIR/models/SadTalker_V0.0.2_256.safetensors" "$SADTALKER_DIR/checkpoints/" || true
ln -sf "$BACKEND_DIR/models/SadTalker_V0.0.2_512.safetensors" "$SADTALKER_DIR/checkpoints/" || true
ln -sf "$BACKEND_DIR/models/mapping_00109-model.pth.tar" "$SADTALKER_DIR/checkpoints/" || true
ln -sf "$BACKEND_DIR/models/mapping_00229-model.pth.tar" "$SADTALKER_DIR/checkpoints/" || true

ln -sf "$BACKEND_DIR/models/codeformer.pth" "$ROOT_DIR/CodeFormer/weights/CodeFormer/" || true

# =========================
# Cleanup
# =========================
echo "==> Cleaning temp files"
rm -rf "$SADTALKER_DIR/checkpoints/__MACOSX" 2>/dev/null || true

# =========================
# Patch basicsr
# =========================
echo "==> Patching basicsr"
cp -v "$ROOT_DIR/CodeFormer/basicsr/archs/"*arch.py \
"$BACKEND_DIR/.venv/lib/python3.11/site-packages/basicsr/archs/" \
2>/dev/null || true

# =========================
# Frontend setup
# =========================
echo "==> Frontend setup"
cd "$FRONTEND_DIR"

if [ ! -d "node_modules" ]; then
  npm install
else
  echo "✓ Frontend already installed"
fi

# =========================
# ENV
# =========================
export HF_HUB_OFFLINE=1
export TORCH_HOME="$BACKEND_DIR/models"
export PYTHONPATH="$BACKEND_DIR"

# =========================
# Run backend
# =========================
echo "==> Starting backend"
cd "$BACKEND_DIR"
cat /dev/null > "$ROOT_DIR/backend.log"
nohup "${PY_RUN[@]}" app.py >> "$ROOT_DIR/backend.log" 2>&1 &

# =========================
# Run frontend
# =========================
echo "==> Starting frontend"
cd "$FRONTEND_DIR"

nohup npm run dev -- --host 0.0.0.0 --port $FRONTEND_PORT > "$ROOT_DIR/frontend.log" 2>&1 &

# =========================
# Wait for backend
# =========================
echo "==> Waiting for backend..."

MAX_RETRIES=24
COUNT=0

until curl -s "http://localhost:$BACKEND_PORT/api/health" > /dev/null; do
    sleep 5
    COUNT=$((COUNT + 1))
    echo "  ...waiting ($((COUNT * 5))s)"

    if [ $COUNT -ge $MAX_RETRIES ]; then
        echo "⚠️ Backend not ready (check logs)"
        break
    fi
done

echo "==> DONE"
echo "Frontend: http://localhost:$FRONTEND_PORT"
echo "Backend : http://localhost:$BACKEND_PORT"
echo "Logs    : backend.log | frontend.log"
