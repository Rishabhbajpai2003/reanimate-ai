#!/usr/bin/env bash
set -euo pipefail

# Run this file from anywhere; paths are resolved relative to this script.

ROOT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
BACKEND_DIR="$ROOT_DIR/backend"
FRONTEND_DIR="$ROOT_DIR/frontend"
BACKEND_PORT=5000
FRONTEND_PORT=3000

echo "==> Project root: $ROOT_DIR"

echo "==> Killing any process on ports $BACKEND_PORT and $FRONTEND_PORT (if running)..."
lsof -ti tcp:$BACKEND_PORT | xargs kill -9 2>/dev/null || true
lsof -ti tcp:$FRONTEND_PORT | xargs kill -9 2>/dev/null || true

echo "==> Backend setup (auto-detect Python environment + deps)"
cd "$BACKEND_DIR"

PY_RUN=()

# 1) Prefer project-local venv if it already exists.
if [ -x ".venv/bin/python" ]; then
  PY_RUN=("$BACKEND_DIR/.venv/bin/python")
  echo "==> Using existing backend venv: $BACKEND_DIR/.venv"

# 2) Otherwise, use currently active Conda/Mamba env if present.
elif [ -n "${CONDA_PREFIX:-}" ] && command -v python >/dev/null 2>&1; then
  PY_RUN=(python)
  echo "==> Using active Conda/Mamba environment: ${CONDA_DEFAULT_ENV:-$CONDA_PREFIX}"

# 3) Otherwise create a local .venv using python3.11, falling back to python3.
else
  if command -v python3.11 >/dev/null 2>&1; then
    BASE_PY=python3.11
  elif command -v python3 >/dev/null 2>&1; then
    BASE_PY=python3
  else
    echo "ERROR: No suitable Python found (need python3.11 or python3)."
    echo "Tip: activate your Conda/Mamba env first, or install Python 3."
    exit 1
  fi

  echo "==> Creating backend venv at $BACKEND_DIR/.venv using $BASE_PY"
  "$BASE_PY" -m venv .venv
  PY_RUN=("$BACKEND_DIR/.venv/bin/python")
fi

echo "==> Installing backend deps"
"${PY_RUN[@]}" -m pip install --upgrade pip
"${PY_RUN[@]}" -m pip install -r ../requirements.txt

echo "==> Ensuring colorization models are available"
"${PY_RUN[@]}" download_models.py --colorize

echo "==> Frontend setup (npm deps)"
cd "$FRONTEND_DIR"
npm install

echo "==> Starting backend on http://localhost:$BACKEND_PORT"
cd "$BACKEND_DIR"
nohup "${PY_RUN[@]}" app.py > "$ROOT_DIR/backend.log" 2>&1 &

echo "==> Starting frontend on http://localhost:$FRONTEND_PORT"
cd "$FRONTEND_DIR"
nohup npm run dev -- --host 0.0.0.0 --port $FRONTEND_PORT > "$ROOT_DIR/frontend.log" 2>&1 &

sleep 3

echo "==> Health check backend"
curl -s "http://localhost:$BACKEND_PORT/api/health" || true
echo

echo "==> Done."
echo "Frontend: http://localhost:$FRONTEND_PORT"
echo "Backend : http://localhost:$BACKEND_PORT"
echo "Logs    : $ROOT_DIR/backend.log and $ROOT_DIR/frontend.log"
